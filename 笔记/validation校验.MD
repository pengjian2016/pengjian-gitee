

@null

@NotNull

@AssertTrue
        布尔 True
@AssertFalse

@Min(val)
        数字 最小val
@Max(val)

@Size(max,min)

@Digits(integer,fraction)

@Past

@PastOrPresent
        过去的日期或者现在
@Future
        将来的日期
@FutureOrPresent
        将来的日期 或者现在
@Pattern(val)

@Email

@NotEmpty
        字节，集合，Map,Array 不能空
@NotBlank
        字符串 不能null
@Positive
        数字-正数
@PositiveOrZero
        数字-正数、0
